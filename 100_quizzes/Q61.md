```.py
def vowels(msg:str):
    x = 'aiueo'
    output = ''
    for letter in msg:
        if letter in x:
            for i in range(5):
                if x[i] == letter:
                    output += str(i+1)
        else:
            output += letter
    return output

print(vowels("Hello World"))
print(vowels("eunoia"))
```
